   const ITEMS_PER_PAGE = 3;

   function renderFreelancers(freelancers, page = 1) {
     if (freelancers.length === 0) {
       freelancersList.innerHTML = '<p class="text-center text-gray-600">Không tìm thấy freelancer phù hợp.</p>';
       pagination.innerHTML = '';
       return;
     }
     const start = (page - 1) * ITEMS_PER_PAGE;
     const end = start + ITEMS_PER_PAGE;
     const pageFreelancers = freelancers.slice(start, end);

     freelancersList.innerHTML = pageFreelancers.map(f => {
       const skillsHtml = f.skills.map(skill => `<span class="bg-indigo-100 text-indigo-700 text-xs font-semibold px-2.5 py-0.5 rounded">${skill}</span>`).join(' ');
       const priceText = f.priceType === "hourly" ? `Giá: $${f.priceMin} - $${f.priceMax} / giờ` : `Giá: $${f.priceMin} - $${f.priceMax}`;
       const starsFull = Math.floor(f.rating);
       const halfStar = f.rating - starsFull >= 0.5;
       const starsEmpty = 5 - starsFull - (halfStar ? 1 : 0);
       const starsHtml = `
         ${'<i class="fas fa-star"></i>'.repeat(starsFull)}
         ${halfStar ? '<i class="fas fa-star-half-alt"></i>' : ''}
         ${'<i class="far fa-star"></i>'.repeat(starsEmpty)}
       `;
       return `
         <article class="bg-white rounded-lg shadow p-6 flex flex-col sm:flex-row items-center sm:items-start space-y-4 sm:space-y-0 sm:space-x-6 hover:shadow-lg transition">
           <img alt="Ảnh đại diện của ${f.name}, freelancer lĩnh vực ${f.field}, kỹ năng ${f.skills.join(', ')}" class="w-28 h-28 rounded-full object-cover flex-shrink-0" height="112" src="${f.avatar}" width="112"/>
           <div class="flex-grow text-center sm:text-left">
             <h3 class="text-2xl font-semibold text-indigo-800 mb-1">${f.name}</h3>
             <p class="text-indigo-600 font-medium mb-2">${f.title}</p>
             <div class="flex flex-wrap gap-2 mb-2 justify-center sm:justify-start">${skillsHtml}</div>
             <div class="flex items-center justify-center sm:justify-start space-x-2 text-yellow-400 mb-2" aria-label="Đánh giá ${f.rating} sao">
               ${starsHtml}
               <span class="text-gray-600 text-sm">(${f.rating.toFixed(1)})</span>
             </div>
             <p class="text-gray-700 mb-2 line-clamp-3">${f.description}</p>
             <div class="text-indigo-600 font-semibold">${priceText}</div>
           </div>
           <div class="flex-shrink-0">
             <a href="${f.url}" class="inline-block mt-2 sm:mt-0 bg-indigo-600 text-white font-semibold rounded-md px-6 py-2 hover:bg-indigo-700 transition" aria-label="Xem hồ sơ freelancer ${f.name}">
               Xem hồ sơ
             </a>
           </div>
         </article>
       `;
     }).join('');

     renderPagination(freelancers.length, page);
   }

   function renderPagination(totalItems, currentPage) {
     const totalPages = Math.ceil(totalItems / ITEMS_PER_PAGE);
     if (totalPages <= 1) {
       pagination.innerHTML = '';
       return;
     }
     let html = '';

     // Previous button
     html += `<button aria-label="Trang trước" class="px-3 py-1 rounded-md border border-indigo-600 text-indigo-600 hover:bg-indigo-600 hover:text-white transition" ${currentPage === 1 ? 'disabled' : ''} type="button" data-page="${currentPage - 1}">
       <i class="fas fa-chevron-left"></i>
     </button>`;

     // Page numbers (show max 5 pages with current in center)
     let startPage = Math.max(1, currentPage - 2);
     let endPage = Math.min(totalPages, currentPage + 2);
     if (currentPage <= 2) {
       endPage = Math.min(5, totalPages);
     }
     if (currentPage >= totalPages - 1) {
       startPage = Math.max(1, totalPages - 4);
     }

     for (let i = startPage; i <= endPage; i++) {
       html += `<button class="px-3 py-1 rounded-md ${i === currentPage ? 'bg-indigo-600 text-white font-semibold' : 'border border-indigo-600 text-indigo-600 hover:bg-indigo-600 hover:text-white transition'}" type="button" data-page="${i}">${i}</button>`;
     }

     // Next button
     html += `<button aria-label="Trang tiếp theo" class="px-3 py-1 rounded-md border border-indigo-600 text-indigo-600 hover:bg-indigo-600 hover:text-white transition" ${currentPage === totalPages ? 'disabled' : ''} type="button" data-page="${currentPage + 1}">
       <i class="fas fa-chevron-right"></i>
     </button>`;

     pagination.innerHTML = html;
   }

   pagination.addEventListener('click', e => {
     if (e.target.tagName === 'BUTTON') {
       const page = Number(e.target.getAttribute('data-page'));
       if (!isNaN(page) && page >= 1 && page <= Math.ceil(filteredFreelancers.length / ITEMS_PER_PAGE)) {
         currentPage = page;
         renderFreelancers(filteredFreelancers, currentPage);
         window.scrollTo({ top: 0, behavior: 'smooth' });
       }
     }
   });

   function normalizeString(str) {
     return str.normalize("NFD").replace(/[\u0300-\u036f]/g, "").toLowerCase();
   }

     const priceMin = parseFloat(filterForm['price-min'].value);
     const priceMax = parseFloat(filterForm['price-max'].value);
     const priceTypes = Array.from(filterForm.querySelectorAll('input[name="price-type"]:checked')).map(i => i.value);
     const ratingMin = parseFloat(filterForm['rating-min'].value);

     filteredFreelancers = freelancersData.filter(f => {
       // Từ khóa: tìm trong tên, tiêu đề, mô tả, bỏ dấu, không phân biệt hoa thường
       if (keyword) {
         const name = normalizeString(f.name);
         const title = normalizeString(f.title);
         const description = normalizeString(f.description);
         if (!name.includes(keyword) && !title.includes(keyword) && !description.includes(keyword)) {
           return false;
         }
       }
       // Lĩnh vực: nếu chọn thì phải có trong danh sách
       if (selectedFields.length > 0 && !selectedFields.includes(f.field)) {
         return false;
       }
       // Kỹ năng: nếu chọn thì freelancer phải có ít nhất 1 trong các kỹ năng được chọn (OR logic)
       if (selectedSkills.length > 0) {
         const hasSkill = selectedSkills.some(skill => f.skills.includes(skill));
         if (!hasSkill) {
           return false;
         }
       }
       // Giá: lọc theo khoảng giá, xét cả min và max của freelancer
       if (!isNaN(priceMin) && f.priceMax < priceMin) {
         return false;
       }
       if (!isNaN(priceMax) && f.priceMin > priceMax) {
         return false;
       }
       // Loại giá: phải nằm trong các loại được chọn
       if (priceTypes.length > 0 && !priceTypes.includes(f.priceType)) {
         return false;
       }
       // Đánh giá tối thiểu
       if (!isNaN(ratingMin) && f.rating < ratingMin) {
         return false;
       }
       return true;
     });

     currentPage = 1;
     renderFreelancers(filteredFreelancers, currentPage);
   }

   filterForm.addEventListener('submit', () => {
     applyFilters();
   });

   clearFiltersBtn.addEventListener('click', () => {
     filterForm.reset();

     // Đặt lại checkbox loại giá về checked (mặc định)
     const priceTypeCheckboxes = filterForm.querySelectorAll('input[name="price-type"]');
     priceTypeCheckboxes.forEach(cb => cb.checked = true);

     // Xóa chọn tất cả các option multiple select
     const multiSelects = [filterForm.field, filterForm.skills];
     multiSelects.forEach(select => {
       for (let i = 0; i < select.options.length; i++) {
         select.options[i].selected = false;
       }
     });

     applyFilters();
   });

   // Khởi tạo danh sách freelancer
   renderFreelancers(freelancersData, 1);
  </script>
